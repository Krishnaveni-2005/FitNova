Primary Key: user_id
Foreign Keys: auth_provider, oauth_provider_id, assigned_trainer_id
No Field Name Data Type (Size) Key Constraints Description
1 user_id INT(11) Primary Key Unique ID for user

9 oauth_provider_id VARCHAR(255) Foreign Key Foreign key reference
10 is_email_verified TINYINT(1) User's email address
11 account_status ENUM('ACTIVE','INACTIVE','SUSPENDED','PENDING') Current status (e.g. Active, Pending)
12 trainer_specialization VARCHAR(100) Trainer specialization
13 trainer_experience INT(11) Trainer experience
14 trainer_certification VARCHAR(255) Trainer certification
15 assigned_trainer_id INT(11) Foreign Key Foreign Key linking to Trainer
16 created_at TIMESTAMP Timestamp / Date of action
17 updated_at TIMESTAMP Timestamp / Date of action
18 assignment_status ENUM('NONE','PENDING','APPROVED','REJECTED','TRAINER_INVITE','LOOKING_FOR_TRAINER') Current status (e.g. Active, Pending)
19 gym_membership_status ENUM('ACTIVE','INACTIVE') Current status (e.g. Active, Pending)
20 profile_picture VARCHAR(255) Profile picture
21 bio TEXT Bio
22 trainer_type ENUM('ONLINE','OFFLINE') Trainer type
